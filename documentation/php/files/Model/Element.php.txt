class Element extends Model{

    private $_debat=null;

    public function getDebat(bool $force=false):Debat
    {
        if($this->_debat==null || $force){
            $this->_debat=Debat::where('id', '=', $this->getAttribute("debat_id"))->get()[0];
        }

        return $this->_debat;
    }
